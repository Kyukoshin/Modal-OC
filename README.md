## Install

```bash
npm install --save modal-example
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'modal-example'
import 'modal-example/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
